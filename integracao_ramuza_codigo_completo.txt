   107	  void addProduct(Product product, {double quantity = 1}) {
   108	    if (quantity <= 0) return;
   109	    if (product.stockQuantity <= 0) return;
   110	
   111	    final index = _items.indexWhere(
   112	      (item) => item.product.id == product.id,
   113	    );
   114	
   115	    if (index == -1) {
   116	      final safeQuantity =
   117	          quantity > product.stockQuantity ? product.stockQuantity : quantity;
   118	
   119	      _items.add(
   120	        SaleCartItem(
   121	          product: product,
   122	          quantity: safeQuantity,
   123	        ),
   124	      );
   125	    } else {
   126	      final current = _items[index];
   127	      final newQuantity = current.quantity + quantity;
   128	      final safeQuantity =
   129	          newQuantity > product.stockQuantity ? product.stockQuantity : newQuantity;

    58	class RamuzaBarcodeParser {
    59	  RamuzaBarcodeParser._();
    60	
    61	  static RamuzaParsedBarcode? tryParse(
    62	    String input,
    63	    RamuzaBarcodeSettings settings,
    64	  ) {
    65	    if (!settings.enabled) return null;
    66	
    67	    final raw = input.trim();
    68	    final digits = raw.replaceAll(RegExp(r'[^0-9]'), '');
    69	
    70	    if (digits.isEmpty) return null;
    71	
    72	    if (digits.length != settings.expectedLength) {
    73	      return null;
    74	    }
    75	
    76	    if (settings.prefix.isNotEmpty && !digits.startsWith(settings.prefix)) {
    77	      return null;
    78	    }
    79	
    80	    final prefixEnd = settings.prefix.length;
    81	    final productEnd = prefixEnd + settings.productCodeDigits;
    82	    final valueEnd = productEnd + settings.valueDigits;
    83	
    84	    if (digits.length < valueEnd) return null;
    85	
    86	    final productCode = digits.substring(prefixEnd, productEnd);
    87	    final valueRaw = digits.substring(productEnd, valueEnd);
    88	    final valueInt = int.tryParse(valueRaw);
    89	
    90	    if (valueInt == null) return null;
    91	
    92	    if (settings.valueMode == RamuzaBarcodeValueMode.weight) {
    93	      return RamuzaParsedBarcode(
    94	        raw: raw,
    95	        digits: digits,
    96	        productCode: productCode,
    97	        valueRaw: valueRaw,
    98	        mode: settings.valueMode,
    99	        quantity: _applyDecimals(valueInt, settings.weightDecimals),
   100	      );
   101	    }
   102	
   103	    return RamuzaParsedBarcode(
   104	      raw: raw,
   105	      digits: digits,
   106	      productCode: productCode,
   107	      valueRaw: valueRaw,
   108	      mode: settings.valueMode,
   109	      totalPrice: _applyDecimals(valueInt, settings.priceDecimals),
   110	    );
   111	  }
   112	
   113	  static String explain(
   114	    String input,
   115	    RamuzaBarcodeSettings settings,
   116	  ) {
   117	    final raw = input.trim();
   118	    final digits = raw.replaceAll(RegExp(r'[^0-9]'), '');
   119	
   120	    if (digits.isEmpty) {
   121	      return 'Digite ou leia um código para testar.';
   122	    }
   123	
   124	    if (!settings.enabled) {
   125	      return 'Leitura de etiqueta Ramuza está desativada.';
   126	    }
   127	
   128	    if (digits.length != settings.expectedLength) {
   129	      return 'Tamanho inválido. Esperado: ${settings.expectedLength} dígitos. Recebido: ${digits.length}.';
   130	    }
   131	
   132	    if (settings.prefix.isNotEmpty && !digits.startsWith(settings.prefix)) {
   133	      return 'Prefixo diferente. Esperado começar com "${settings.prefix}".';
   134	    }
   135	
   136	    final parsed = tryParse(input, settings);
   137	
   138	    if (parsed == null) {
   139	      return 'Não foi possível interpretar o código.';
   140	    }
   141	
   142	    if (parsed.mode == RamuzaBarcodeValueMode.weight) {
   143	      return 'Produto/PLU: ${parsed.productCode}\nPeso: ${_formatNumber(parsed.quantity ?? 0)}';
   144	    }
   145	
   146	    return 'Produto/PLU: ${parsed.productCode}\nValor total: ${_formatMoney(parsed.totalPrice ?? 0)}';
   147	  }

   149	  static double _applyDecimals(int value, int decimals) {
   150	    var divisor = 1;
   151	
   152	    for (var i = 0; i < decimals; i++) {
   153	      divisor *= 10;
   154	    }
   155	
   156	    return value / divisor;
   157	  }

   115	      if (seenCodes.containsKey(code)) {
   116	        errors.add(
   117	          'Código duplicado: $code usado em "${seenCodes[code]}" e "${product.name}".',
   118	        );
   119	      } else {
   120	        seenCodes[code] = product.name;
   121	      }
   122	    }
   123	
   124	    return RamuzaExportValidation(
   125	      errors: errors,
   126	      warnings: warnings,
   127	    );
   128	  }

   130	  static RamuzaExportFile buildFile({
   131	    required List<Product> products,
   132	    required RamuzaExportFormat format,
   133	    required int validityDays,
   134	    required bool removeAccents,
   135	  }) {
   136	    final content = switch (format) {
   137	      RamuzaExportFormat.pluCsvHeader => _buildPluCsv(
   138	          products: products,
   139	          validityDays: validityDays,
   140	          removeAccents: removeAccents,
   141	          includeHeader: true,
   142	          separator: ';',
   143	        ),
   144	      RamuzaExportFormat.pluCsvNoHeader => _buildPluCsv(
   145	          products: products,
   146	          validityDays: validityDays,
   147	          removeAccents: removeAccents,
   148	          includeHeader: false,
   149	          separator: ';',
   150	        ),
   151	      RamuzaExportFormat.pluTxtTab => _buildPluCsv(
   152	          products: products,
   153	          validityDays: validityDays,
   154	          removeAccents: removeAccents,
   155	          includeHeader: false,
   156	          separator: '\t',
   157	        ),
   158	      RamuzaExportFormat.simpleCsv => _buildSimpleCsv(
   159	          products: products,
   160	          validityDays: validityDays,
   161	          removeAccents: removeAccents,
   162	        ),
   163	      RamuzaExportFormat.priceOnlyCsv => _buildPriceOnlyCsv(
   164	          products: products,
   165	          removeAccents: removeAccents,
   166	        ),
   167	    };
   168	
   169	    return RamuzaExportFile(
   170	      format: format,
   171	      fileName: format.fileName,
   172	      content: content,
   173	    );
   174	  }

   331	  static String _pluNumber(Product product, int index) {
   332	    final code = _digitsOnly(product.code);
   333	    final parsed = int.tryParse(code);
   334	
   335	    if (parsed != null && parsed > 0) {
   336	      return parsed.toString();
   337	    }
   338	
   339	    return (index + 1).toString();
   340	  }

   342	  static String _scaleUnit(ProductUnit unit) {
   343	    switch (unit) {
   344	      case ProductUnit.kg:
   345	        return 'Peso';
   346	      default:
   347	        return 'Peca';
   348	    }

   351	  static String _formatMoney(double value) {
   352	    return value.toStringAsFixed(2).replaceAll('.', ',');
   353	  }
   354	
   355	  static String _digitsOnly(String value) {
   356	    return value.replaceAll(RegExp(r'[^0-9]'), '');
   357	  }
   358	
   359	  static String _cleanText(
   360	    String value, {
   361	    required bool removeAccents,
   362	    required int max,
   363	  }) {
   364	    var text = value.trim();
   365	
   366	    if (removeAccents) {
   367	      text = _removeAccents(text);
   368	    }
   369	
   370	    text = text
   371	        .replaceAll(';', ' ')
   372	        .replaceAll('\t', ' ')
   373	        .replaceAll('\n', ' ')
   374	        .replaceAll('\r', ' ')
   375	        .replaceAll(RegExp(r'\s+'), ' ')
   376	        .toUpperCase();
   377	
   378	    if (text.length > max) {
   379	      text = text.substring(0, max);
   380	    }
   381	
   382	    return text;
   383	  }
   384	
   385	  static String _removeAccents(String value) {
   386	    const from = 'ÁÀÂÃÄÉÈÊËÍÌÎÏÓÒÔÕÖÚÙÛÜÇÑáàâãäéèêëíìîïóòôõöúùûüçñ';
   387	    const to = 'AAAAAEEEEIIIIOOOOOUUUUCNaaaaaeeeeiiiiooooouuuucn';
   388	
   389	    var result = value;
   390	
   391	    for (var i = 0; i < from.length; i++) {
   392	      result = result.replaceAll(from[i], to[i]);
   393	    }
   394	
   395	    return result;
   396	  }

   437	Product? _findProductByRamuzaCode(List<Product> products, String code) {
   438	  final normalizedCode = _normalizeNumericCode(code);
   439	
   440	  for (final product in products) {
   441	    if (product.isDeleted) continue;
   442	
   443	    final productCode = _normalizeNumericCode(product.code);
   444	
   445	    if (productCode == normalizedCode) {
   446	      return product;
   447	    }
   448	  }
   449	
   450	  return null;
   451	}

   453	String _normalizeNumericCode(String value) {
   454	  final digits = value.replaceAll(RegExp(r'[^0-9]'), '');
   455	  final parsed = int.tryParse(digits);
   456	
   457	  if (parsed == null) return digits;
   458	
   459	  return parsed.toString();
   460	}

   462	int _safeInt(String value, {required int fallback}) {
   463	  final parsed = int.tryParse(value.trim());
   464	
   465	  if (parsed == null || parsed < 0) {
   466	    return fallback;
   467	  }
   468	
   469	  return parsed;
   470	}

   473	  if (value % 1 == 0) {

   475	  }

   230	  void _saveProduct() {
   231	    final inventory = context.read<InventoryProvider>();
   232	
   233	    final code = codeController.text.trim();
   234	    final name = nameController.text.trim();
   235	    final salePrice = _parseMoney(salePriceController.text);
   236	    final costPrice = _parseMoney(costPriceController.text);
   237	    final stock = _parseNumber(stockController.text);
   238	    final minStock = _parseNumber(minStockController.text);
   239	
   240	    if (code.isEmpty) {
   241	      _showMessage('Código inválido.');
   242	      return;
   243	    }
   244	
   245	    if (name.isEmpty) {
   246	      _showMessage('Informe o nome do produto.');
   247	      return;
   248	    }
   249	
   250	    if (salePrice <= 0) {
   251	      _showMessage('Informe o preço de venda.');
   252	      return;
   253	    }
   254	
   255	    if (stock <= 0) {
   256	      _showMessage('Informe o estoque inicial.');
   257	      return;
   258	    }
   259	
   260	    final now = DateTime.now();
   261	
   262	    final product = Product(
   263	      id: now.microsecondsSinceEpoch.toString(),
   264	      code: code,
   265	      name: name,
   266	      category: category,
   267	      unit: unit,
   268	      salePrice: salePrice,
   269	      costPrice: costPrice,
   270	      stockQuantity: stock,
   271	      minStock: minStock,
   272	      favorite: false,
   273	      imagePath: null,
   274	      createdAt: now,
   275	      updatedAt: now,
   276	    );
   277	
   278	    inventory.addProduct(product);
   279	
   280	    Navigator.of(context).pop(product);
   281	  }

   292	double _parseMoney(String value) {
   293	  var text = value.trim().replaceAll('R\$', '').replaceAll(' ', '');
   294	
   295	  if (text.contains(',')) {
   296	    text = text.replaceAll('.', '').replaceAll(',', '.');
   297	  }
   298	
   299	  return double.tryParse(text) ?? 0;
   300	}

   303	  final text = value.trim().replaceAll(',', '.');

   669	Future<void> _handleSalesInput(BuildContext context, String value) async {
   670	  final inventory = context.read<InventoryProvider>();
   671	  final sales = context.read<SalesProvider>();
   672	  final ramuzaSettings = context.read<RamuzaSettingsProvider>().settings;
   673	
   674	  final parsed = RamuzaBarcodeParser.tryParse(value, ramuzaSettings);
   675	
   676	  if (parsed != null) {
   677	    var product = _findByRamuzaCode(inventory.products, parsed.productCode);
   678	
   679	    if (product == null) {
   680	      _showMessage(
   681	        context,
   682	        'Etiqueta lida. PLU ${parsed.productCode} ainda não está cadastrado.',
   683	      );
   684	
   685	      product = await showQuickProductFromBarcodeDialog(
   686	        context: context,
   687	        productCode: parsed.productCode,
   688	        rawBarcode: parsed.digits,
   689	        suggestedQuantity: parsed.quantity,
   690	      );
   691	
   692	      if (product == null) {
   693	        _logRamuzaBarcodeRead(
   694	          context: context,
   695	          parsed: parsed,
   696	          product: null,
   697	          status: RamuzaBarcodeStatus.canceledQuickRegister,
   698	          message: 'PLU ${parsed.productCode} não cadastrado. Cadastro rápido cancelado.',
   699	          screen: 'PDV',
   700	        );
   701	
   702	        sales.setSearchTerm('');
   703	        return;
   704	      }
   705	    }
   706	
   707	    if (product.isDeleted) {
   708	      _logRamuzaBarcodeRead(
   709	        context: context,
   710	        parsed: parsed,
   711	        product: product,
   712	        status: RamuzaBarcodeStatus.productDeleted,
   713	        message: 'Produto está na lixeira.',
   714	        screen: 'PDV',
   715	      );
   716	
   717	      _showMessage(
   718	        context,
   719	        'Etiqueta lida. Produto ${product.name} está na lixeira.',
   720	      );
   721	
   722	      sales.setSearchTerm('');
   723	      return;
   724	    }
   725	
   726	    final quantity = parsed.quantityForProduct(product);
   727	
   728	    if (quantity == null || quantity <= 0) {
   729	      _logRamuzaBarcodeRead(
   730	        context: context,
   731	        parsed: parsed,
   732	        product: product,
   733	        status: RamuzaBarcodeStatus.invalidQuantity,
   734	        message: 'Não foi possível calcular a quantidade/valor.',
   735	        screen: 'PDV',
   736	      );
   737	
   738	      _showMessage(
   739	        context,
   740	        'Etiqueta lida, mas não deu para calcular quantidade/valor.',
   741	      );
   742	
   743	      sales.setSearchTerm('');
   744	      return;
   745	    }
   746	
   747	    if (product.stockQuantity <= 0) {
   748	      _logRamuzaBarcodeRead(
   749	        context: context,
   750	        parsed: parsed,
   751	        product: product,
   752	        status: RamuzaBarcodeStatus.stockEmpty,
   753	        message: 'Estoque zerado.',
   754	        screen: 'PDV',
   755	        quantity: quantity,
   756	      );
   757	
   758	      _showMessage(
   759	        context,
   760	        'Etiqueta lida: ${product.name}, mas o estoque está zerado.',
   761	      );
   762	
   763	      sales.setSearchTerm('');
   764	      return;
   765	    }
   766	
   767	    final error = _validateQuantity(product, quantity);
   768	
   769	    if (error != null) {
   770	      _logRamuzaBarcodeRead(
   771	        context: context,
   772	        parsed: parsed,
   773	        product: product,
   774	        status: RamuzaBarcodeStatus.invalidQuantity,
   775	        message: error,
   776	        screen: 'PDV',
   777	        quantity: quantity,
   778	      );
   779	
   780	      _showMessage(
   781	        context,
   782	        'Etiqueta lida: ${product.name}. $error',
   783	      );
   784	
   785	      sales.setSearchTerm('');
   786	      return;
   787	    }
   788	
   789	    sales.addProduct(product, quantity: quantity);
   790	    sales.setSearchTerm('');
   791	
   792	    _logRamuzaBarcodeRead(
   793	      context: context,
   794	      parsed: parsed,
   795	      product: product,
   796	      status: RamuzaBarcodeStatus.success,
   797	      message: 'Produto adicionado ao carrinho.',
   798	      screen: 'PDV',
   799	      quantity: quantity,
   800	    );
   801	
   802	    _showMessage(
   803	      context,
   804	      'Etiqueta OK: ${product.name} • ${_formatNumber(quantity)} ${product.unit.label}.',
   805	    );
   806	
   807	    return;
   808	  }
   809	
   810	  final product = _findByCodeOrName(inventory.products, value);
   811	
   812	  if (product == null) {
   813	    _showMessage(context, 'Produto não encontrado.');
   814	    return;
   815	  }
   816	
   817	  sales.addProduct(product);
   818	  sales.setSearchTerm('');
   819	
   820	  _showMessage(
   821	    context,
   822	    '${product.name} adicionado com 1 ${product.unit.label}.',
   823	  );
   824	}

   826	void _logRamuzaBarcodeRead({
   827	  required BuildContext context,
   828	  required RamuzaParsedBarcode parsed,

   296	
   297	  Future<void> _handleSubmitted(String value) async {
   298	    final text = value.trim();
   299	
   300	    if (text.isEmpty) {
   301	      _clearScanner();
   302	      return;
   303	    }
   304	
   305	    final inventory = context.read<InventoryProvider>();
   306	    final sales = context.read<SalesProvider>();
   307	    final ramuzaSettings = context.read<RamuzaSettingsProvider>().settings;
   308	
   309	    final parsed = RamuzaBarcodeParser.tryParse(text, ramuzaSettings);
   310	
   311	    if (parsed != null) {
   312	      var product = _findByRamuzaCode(inventory.products, parsed.productCode);
   313	
   314	      if (product == null) {
   315	        _showMessage(
   316	          context,
   317	          'PLU ${parsed.productCode} não cadastrado. Abrindo cadastro rápido.',
   318	        );
   319	
   320	        product = await showQuickProductFromBarcodeDialog(
   321	          context: context,
   322	          productCode: parsed.productCode,
   323	          rawBarcode: parsed.digits,
   324	          suggestedQuantity: parsed.quantity,
   325	        );
   326	
   327	        if (!mounted) return;
   328	
   329	        if (product == null) {
   330	          _logRamuzaBarcodeRead(
   331	            context: context,
   332	            parsed: parsed,
   333	            product: null,
   334	            status: RamuzaBarcodeStatus.canceledQuickRegister,
   335	            message: 'Cadastro rápido cancelado.',
   336	          );
   337	
   338	          _clearScanner();
   339	          return;
   340	        }
   341	      }
   342	
   343	      if (product.isDeleted) {
   344	        _logRamuzaBarcodeRead(
   345	          context: context,
   346	          parsed: parsed,
   347	          product: product,
   348	          status: RamuzaBarcodeStatus.productDeleted,
   349	          message: 'Produto está na lixeira.',
   350	        );
   351	
   352	        _showMessage(context, '${product.name} está na lixeira.', error: true);
   353	        _clearScanner();
   354	        return;
   355	      }
   356	
   357	      final quantity = parsed.quantityForProduct(product);
   358	
   359	      if (quantity == null || quantity <= 0) {
   360	        _logRamuzaBarcodeRead(
   361	          context: context,
   362	          parsed: parsed,
   363	          product: product,
   364	          status: RamuzaBarcodeStatus.invalidQuantity,
   365	          message: 'Quantidade inválida na etiqueta.',
   366	        );
   367	
   368	        _showMessage(context, 'Quantidade inválida na etiqueta.', error: true);
   369	        _clearScanner();
   370	        return;
   371	      }
   372	
   373	      final validation = _validateQuantity(product, quantity);
   374	
   375	      if (validation != null) {
   376	        _logRamuzaBarcodeRead(
   377	          context: context,
   378	          parsed: parsed,
   379	          product: product,
   380	          status: RamuzaBarcodeStatus.invalidQuantity,
   381	          message: validation,
   382	          quantity: quantity,
   383	        );
   384	
   385	        _showMessage(context, validation, error: true);
   386	        _clearScanner();
   387	        return;
   388	      }
   389	
   390	      sales.addProduct(product, quantity: quantity);
   391	
   392	      _logRamuzaBarcodeRead(
   393	        context: context,
   394	        parsed: parsed,
   395	        product: product,
   396	        status: RamuzaBarcodeStatus.success,
   397	        message: 'Etiqueta adicionada ao carrinho.',
   398	        quantity: quantity,
   399	      );
   400	
   401	      _showMessage(
   402	        context,
   403	        'Etiqueta OK: ${product.name} • ${_formatNumber(quantity)} ${product.unit.label}.',
   404	        success: true,
   405	      );
   406	
   407	      _clearScanner();
   408	      return;
   409	    }
   410	
   411	    final product = _findByCodeOrName(inventory.products, text);
   412	
   413	    if (product == null) {
   414	      _showMessage(context, 'Produto não encontrado.', error: true);
   415	      _clearScanner();
   416	      return;
   417	    }
   418	
   419	    _addProduct(product);
   420	    _clearScanner();
   421	  }
   422	
   423	  void _addProduct(Product product) {
   424	    final validation = _validateQuantity(product, 1);
   425	
   426	    if (validation != null) {
   427	      _showMessage(context, validation, error: true);
   428	      _refocusScanner();
   429	      return;
   430	    }
   431	
   432	    context.read<SalesProvider>().addProduct(product);
   433	
   434	    _showMessage(
   435	      context,
   436	      '${product.name} adicionado.',
   437	    );
   438	
   439	    _refocusScanner();
   440	  }

   934	Future<void> _finishOperationSale(BuildContext context) async {
   935	  final inventory = context.read<InventoryProvider>();
   936	  final sales = context.read<SalesProvider>();
   937	
   938	  if (sales.paymentMethod == PaymentMethod.fiado) {
   939	    sales.setPaymentMethod(PaymentMethod.dinheiro);
   940	  }
   941	
   942	  final validationError = inventory.validateSaleItems(sales.items);
   943	
   944	  if (validationError != null) {
   945	    _showMessage(context, validationError, error: true);
   946	    return;
   947	  }

  1096	  final available = products.where((product) {

  1114	  return available.where((product) {
  1115	    return product.name.toLowerCase().contains(term) ||
  1116	        product.code.toLowerCase().contains(term) ||
