1190: class _SupplierStats {
1191:   const _SupplierStats({
1192:     required this.total,
1193:     required this.open,
1194:     required this.lastPurchase,
1195:   });
1196: 
1197:   final double total;
1198:   final double open;
1199:   final DateTime? lastPurchase;
1200: }
1201: 
1202: _SupplierStats _supplierStats(
1203:   String supplierName,
1204:   List<SupplierPurchase> purchases,
1205: ) {
1206:   final normalized = supplierName.trim().toLowerCase();
1207: 
1208:   final supplierPurchases = purchases.where((purchase) {
1209:     return purchase.supplierName.trim().toLowerCase() == normalized;
1210:   }).toList();
1211: 
1212:   supplierPurchases.sort((a, b) => b.purchaseDate.compareTo(a.purchaseDate));
1213: 
1214:   final total = supplierPurchases.fold<double>(
1215:     0,
1216:     (sum, purchase) => sum + purchase.totalCost,
1217:   );
1218: 
1219:   final open = supplierPurchases.where((purchase) => !purchase.paid).fold<double>(
1220:         0,
1221:         (sum, purchase) => sum + purchase.totalCost,
1222:       );
1223: 
1224:   return _SupplierStats(
1225:     total: total,
1226:     open: open,
1227:     lastPurchase:
1228:         supplierPurchases.isEmpty ? null : supplierPurchases.first.purchaseDate,
1229:   );
1230: }
